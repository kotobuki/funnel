    def process_sample(value)
      @buffer[0] = value
      result = 0.0
      @coef.length.times do |i|
        result = result + @buffer[i] * @coef[i]
      end

      (@buffer.length - 1).downto(1) do |i|
        @buffer[i] = @buffer[i - 1]
      end

      return result
    end
