    def initialize(wave_func, frequency, *args)
      @wave_func = wave_func
      if frequency > 0 then
        @frequency = frequency
      else
        raise ArgumentError, "frequency should be greater than 0"
      end

      if args.length == 4 then
        @amplitude = args[0]
        @offset = args[1]
        @phase = args[2]
        @times = args[3]
      elsif args.length == 1 then
        @amplitude = 1.0
        @offset = 0.0
        @phase = 0.0
        @times = args[0]
      else
        raise ArgumentError, "wrong number of arguments"
      end

      @value = 0.0
      @time = 0.0
      @start_time = 0.0
      @auto_update = false
      @listener = nil
      @last_val = 0.99

      if @@service_thread == nil then
        @@service_thread = Thread.new do
          loop do
            @@listeners.each do |listener|
              listener.send(:update)
            end
            sleep(@@interval)
          end
        end
      end
    end

    def update(*args)
      if args.length == 1 then
        @time = @time + args[0]
      elsif @auto_update then
        @time = Time.now.to_f - @start_time
      else
        @time = @time + @@interval
      end

      if (@times != 0 and @frequency * @time > @times) then
        @value = @offset
        @listener.call(@value) if @listener
        @@listeners.delete(self)
        @time = @times / @frequency
      else
        val = @frequency * (@time + @phase)
        @value = @amplitude * @wave_func.call(val, @last_val) + @offset
        @last_val = val
        @listener.call(@value) if @listener
      end
    end
